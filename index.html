      <script>
      const carousel = document.getElementById("carousel");
      const cards = carousel.querySelectorAll(".card");
      const wrapper = document.getElementById("carouselWrapper");
      const totalCards = cards.length;
      const angleStep = 360 / totalCards;
      let currentIndex = 0;
      let rotationInterval = null;
      let isShrunk = false;
      function positionCards(activeIndex) {
      for (let i = 0; i < totalCards; i++) {
      let angle = i * angleStep - activeIndex * angleStep;
      let card = cards[i];
      let transform = "rotateY(" + angle + "deg) translateZ(300px)";
      let isActive = false;
      // Normal mode: only front card pops
      if (!isShrunk && angle % 360 === 0) {
      isActive = true;
      transform = "rotateY(" + angle + "deg) translateZ(400px) scale(1.3)";
      }
      // Shrink mode: pop front card and 2-away neighbors
      if (isShrunk) {
      const diff = ((i - activeIndex + totalCards) % totalCards);
      if (diff === 0 || diff === 2 || diff === totalCards - 2) {
      isActive = true;
      transform = "rotateY(" + angle + "deg) translateZ(400px) scale(1.3)";
      }
      }
      card.classList.toggle("active", isActive);
      card.style.transform = transform;
      }
      }
      function startRotation() {
      if (!rotationInterval) {
      rotationInterval = setInterval(() => {
      currentIndex = (currentIndex + 1) % totalCards;
      positionCards(currentIndex);
      }, 3000);
      }
      }
      function stopRotation() {
      clearInterval(rotationInterval);
      rotationInterval = null;
      }
      // Initial position and rotation start
      positionCards(currentIndex);
      startRotation();
      // Scroll behavior
      window.addEventListener("scroll", function () {
      const scrollY = window.scrollY;
      if (scrollY > 100) {
      wrapper.classList.add("shrink");
      stopRotation();
      isShrunk = true;
      } else {
      wrapper.classList.remove("shrink");
      startRotation();
      isShrunk = false;
      }
      // Reposition cards on scroll to update pop logic
      positionCards(currentIndex);
      });
      </script>
